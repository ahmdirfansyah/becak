<!-- ----Becak Algorithm---- -->
STORE variable people
DECLARE variable vmax
DECLARE variable vnow
SET vmax = 20

IF people = 0 then
vnow = vmax - 0
else if
people greater than 2 then
vnow = vmax - 5
else if 
people equals to 4 then
vnow = vmax - 7
else
vnow = vmax - 2

DISPLAY vnow
